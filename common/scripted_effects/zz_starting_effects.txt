effect_starting_pop_wealth_usa = {
	every_scope_pop = {
		set_pop_wealth = {
			wealth_distribution = {
				value = 15
				if = {
					limit = { strata = rich }
					add = 20
				}
				if = {
					limit = { strata = middle }
					add = 10
				}
				if = {
					limit = { is_pop_type = laborers }
					add = -5
				}
				if = {
					limit = { pop_employment_building_group = bg_gold_mining }
					add = 5
				}					
				if = {
					limit = { pop_is_discriminated = yes }
					multiply = 0.75
				}
				# TODO: additional penalty when in a colony or territory
			}
			update_loyalties = no
		}
		effect_reset_wealth_if_slave = yes
	}
}

free_unemployed_slaves = {

	is_shown = { has_law = law_type:law_slave_trade }

	possible = {
		any_scope_pop = {
			is_pop_type = slaves
			is_employed = no
		}
	}

	when_taken = {
        every_scope_pop = {
			limit = {
				is_pop_type = slaves
				is_employed = no
			}
			change_poptype = pop_type:laborers
		}
	}

	ai_chance = {
		base = 10
	}
}

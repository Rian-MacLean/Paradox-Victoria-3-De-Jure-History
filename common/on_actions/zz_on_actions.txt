                    is_building_type = building_university
                    level >= 5
                }
			}
			add_modifier = modifier_first_university

            add_modifier = modifier_third_university
            add_modifier = modifier_forth_university
            add_modifier = modifier_fifth_university
		}
        else_if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_university

                }
			}
			add_modifier = modifier_first_university
            add_modifier = modifier_second_university
            add_modifier = modifier_third_university
            add_modifier = modifier_forth_university
            remove_modifier = modifier_fifth_university
        }
        else_if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_university
                    level = 3
                }
			}
			add_modifier = modifier_first_university
            add_modifier = modifier_second_university
            add_modifier = modifier_third_university
            remove_modifier = modifier_forth_university
            remove_modifier = modifier_fifth_university
        }
        else_if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_university
                    level = 2
                }
			}
			add_modifier = modifier_first_university
            add_modifier = modifier_second_university
            remove_modifier = modifier_third_university
            remove_modifier = modifier_forth_university
            remove_modifier = modifier_fifth_university
        }
        else_if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_university
                    level = 1
                }
			}

            remove_modifier = modifier_forth_university
            remove_modifier = modifier_fifth_university

            remove_modifier = modifier_third_university
            remove_modifier = modifier_forth_university

on_acquired_technology = {
	effect = {
		if = {
            limit = {
                has_technology_researched = quinine
                has_modifier = modifier_early_colonization_three
            }
            remove_modifier = modifier_early_colonization_three
        }
		if = {
            limit = {
                has_technology_researched = civilizing_mission
                has_modifier = modifier_early_colonization_four
            }
            remove_modifier = modifier_early_colonization_four
        }
		if = {
            limit = {
                has_technology_researched = malaria_prevention
                has_modifier = modifier_early_colonization_five
            }
            remove_modifier = modifier_early_colonization_five
        }
		if = {
            limit = {
                exists = c:LIB
                root = c:LIB
                has_law = law_type:law_local_police
                NOT = { has_modifier = modifier_local_police } 
            }
            add_modifier = modifier_local_police
        }
        else_if = {
            limit = { has_modifier = modifier_local_police }
            remove_modifier = modifier_local_police
        }
		if = {
            limit = {
                has_technology_researched = central_archives
                has_modifier = modifier_early_security_four
            }
            remove_modifier = modifier_early_security_four
        }
		if = {
            limit = {

                has_modifier = modifier_early_security_five
            }

        }
    }
}
